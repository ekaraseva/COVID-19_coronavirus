# Function to keep all entities for 'France', 'United Kingdom', 'Netherlands', 'Denmark', 
# but use their 'Province/State' name as 'Country/Region'

def data_preparation(data):
    print(data.shape)
    # check country names for duplicates
    print (data[data['Country/Region'].duplicated()]['Country/Region'].value_counts())
    
    # Countries such as France, UK, Netherlands, Denmark have islands territories, let's keep them separate
    for i in ['France', 'United Kingdom', 'Netherlands', 'Denmark']:
        data.loc[data['Country/Region']==i, 'Country/Region']=data[data['Country/Region']==i]['Province/State']
        
    # check country names for duplicates
    #print (data[data['Country/Region'].duplicated()]['Country/Region'].value_counts())       
    
    # groupby by Other duplicated countries
    data=data.groupby(['Country/Region']).sum().reset_index()
    print(data.shape)
    return data

def re_structure(data, text):
    output_df= pd.DataFrame(columns=['Country/Region',text, 'date' ])  
    
    for i in range(1, len(data.columns)):
        subset1=data[['Country/Region',data.columns[i]]]
        subset1['date']=data.columns[i]
        subset1.rename(columns={data.columns[i]: text}, inplace=True)

        output_df=output_df.append(subset1)
        i=i+1
    output_df.rename(columns={'Country/Region':'country'}, inplace=True)
    output_df['date']=pd.to_datetime(output_df['date']).dt.date
    return output_df

# Function to select subset of dataframe per country names 
def select_subset(data, columns_list):
    # make empty df
    subset_df=pd.DataFrame(columns=data.columns)
    for i in columns_list:
        data1=data[data.country==i]
        subset_df=subset_df.append(data1)
    return subset_df
